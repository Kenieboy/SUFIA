// FILTER BY DATE

SELECT 
    pd.DATEDELIVERED,
    pd.PURCHASEDELIVERYNO,
    i.CODE AS ITEMCODE,
    i.NAMEENG AS MATERIALNAME,
    pdd.QTY,
    iu.DESCRIPTIONEN AS UNIT
FROM  PURCHASEDELIVERYDETAIL pdd
LEFT JOIN PURCHASEDELIVERY pd ON pd.ID = pdd.PURCHASEDELIVERYID
LEFT JOIN ITEMVARIATION iv ON pdd.ITEMVARIATIONID = iv.ID
LEFT JOIN  ITEM i ON iv.ITEMID = i.ID
LEFT JOIN  ITEMUNIT iu ON iv.ITEMUNITID = iu.ID
WHERE pd.DATEDELIVERED BETWEEN '2024-1-11' AND '2024-6-30'
ORDER BY pd.DATEDELIVERED DESC
